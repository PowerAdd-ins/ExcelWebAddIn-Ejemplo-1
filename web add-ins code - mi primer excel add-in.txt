'use strict';

(function () {

    // The initialize function is run each time the page is loaded.
    Office.initialize = function (reason) {

        $(document).ready(function () {

            // Use this to check whether the API is supported in the Excel client.
            if (Office.context.requirements.isSetSupported('ExcelApi', 1.1)) {
                
                $('#button1').click(show_Welcome);
                $('#supportedVersion').html('This code is using Excel 2016 or greater.');
            }
            
            else {
                
                $('#supportedVersion').html('This code requires Excel 2016 or greater.');
            
            }
        });
    };

    function show_Welcome() {
        Excel.run(function (context) {

            Excel.run(function (ctx) {
                // Create a proxy object for the active sheet
                var sheet = ctx.workbook.worksheets.getActiveWorksheet();

                sheet.getCell(0, 0).values = "Bienvenidos a nuestro curso de Office Add-ins!!!";
                sheet.getCell(0, 0).worksheet.getUsedRange().format.font.size = 30;

                return ctx.sync();
            })
                .catch(errorHandler);
   
        })
            .catch(function (error) {
                console.log('Error: ' + JSON.stringify(error));
                if (error instanceof OfficeExtension.Error) {
                    console.log('Debug info: ' + JSON.stringify(error.debugInfo));
                }
            });
    }


   

})();
